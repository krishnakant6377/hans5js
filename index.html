Q6. Find the Winner?

        Ans. var Game_Winner = (S) => {
            
          let count = 0;
          let add = 0;
          for(let i = 0; i < S.length; i++)
          {
            if(S[i] == "A")
            {
              count++;
            }
            else{
              add++;
            }
          }
          if(count == add)
          {
            return "Draw";
          }
          else if(count > add){
            return "Aditya";
          }
         else{
            return "Danish";
          }
        };

Q8. Plaindrome Check?

        Ans. var Palin_Check = (str) => {
          const len = str.length;
        
            for (let i = 0; i < len / 2; i++) {
        
                if (str[i] !== str[len - 1 - i]) {
                    return 'False';
                }
            }
            return 'True';
        }
